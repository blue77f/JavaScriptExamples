        function rastPoz(maxYuk,maxGen,birim){
            var yukseklik=Math.round(Math.random()*maxYuk)+birim,genislik=Math.round(Math.random()*maxGen)+birim;
            return [yukseklik,genislik];
        }
        function bicim(){
            var bicim=Math.round(Math.random()*60)+"%";
            return bicim;
            
        }
        function rastRenk(){
            var olasiliklar="0123456789abcdef",renk="#";
            for(var i=0;i<6;i++){
            renk+=olasiliklar.split("")[Math.floor(Math.random()*15)];
                             }
